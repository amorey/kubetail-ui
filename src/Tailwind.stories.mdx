export const rgbToHex = (rgb) => {
  const [r, g, b] = rgb.split(' ');
  return '#' + [r, g, b].map(x => parseInt(x).toString(16).padStart(2, '0')).join('');
}

# Tailwind Configuration

To add kubetail-ui's custom Tailwind theme to your project, add the kubetail-ui Tailwind plugin 
and content folder to your Tailwind config file (tailwind.config.js):

```javascript
/** @type {import('tailwindcss').Config} */
module.exports = {
  content: [
    ...
    "./node_modules/kubetail-ui/esm/**/*.js",
  ],
  ...
  plugins: [
    ...
    require('kubetail-ui/plugin'),
  ],
}
```
